::@echo off
echo ffmpeg-mini-gui
echo -------------------
echo.

::Change code page to Unicode
chcp 65001

set bitrate=*bitrate*
set preset=*preset*
set output=*output*
set filters=*filters*
set ffmpeg=*curr*\ffmpeg.exe

%ffmpeg% -i "*video*"%filters%-b:v %bitrate% %timecut% -preset %preset% -vcodec libx264 -acodec copy "%output%"

@echo.
@echo when no errors - destination video saved to %output%
@echo.
::pause
timeout /t 10
exit
